22  void asset_loader::flip_image_x(void* _data, uint32_t _width, uint32_t _height, uint32_t _bytes_per_pixel) {
23  const uint32_t bytes_per_row = _width * _bytes_per_pixel;
24  uint8_t* temp_pixel = new uint8_t[_bytes_per_pixel];
25 
26  // For every row, we need to flip the first and last pixel.
27  for (uint32_t r = 0; r < _height; ++r) {
28  uint8_t* row = static_cast<uint8_t*>(_data) + (r * bytes_per_row);
29  // Swap the first and last pixel, then the second and second last pixel, then the third and third last pixel...
30  for (uint32_t c = 0; c < (_width >> 1); ++c) {
31  uint8_t* left_pixel = &row[c * _bytes_per_pixel]; // Left Pixel (1st Pixel, 2nd Pixel, 3rd Pixel etc...)
32  uint8_t* right_pixel = &row[(_width - c - 1) * _bytes_per_pixel]; // Right Pixel (Last Pixel, 2nd Last Pixel, 3rd Last Pixel etc...)
33  std::memcpy(temp_pixel, left_pixel, _bytes_per_pixel);
34  std::memcpy(left_pixel, right_pixel, _bytes_per_pixel);
35  std::memcpy(right_pixel, temp_pixel, _bytes_per_pixel);
36  }
37  }
38 
39  delete[] temp_pixel;
40  }

42  void asset_loader::flip_image_y(void* _data, uint32_t _width, uint32_t _height, uint32_t _bytes_per_pixel) {
43  const uint32_t bytes_per_row = _width * _bytes_per_pixel;
44  uint8_t* temp_row = new uint8_t[bytes_per_row];
45 
46  // Since the data is stored in a row-major format, we can swap the data row by row rather than pixel by pixel.
47  for (uint32_t r = 0; r < (_height >> 1); ++r) {
48  uint8_t* top_row = &static_cast<uint8_t*>(_data)[r * bytes_per_row];
49  uint8_t* bottom_row = &static_cast<uint8_t*>(_data)[(_height - r - 1) * bytes_per_row];
50  std::memcpy(temp_row, top_row, bytes_per_row);
51  std::memcpy(top_row, bottom_row, bytes_per_row);
52  std::memcpy(bottom_row, temp_row, bytes_per_row);
53  }

459  positions.push_back(position);
460  }
461 
462  // TexCoord
463  if (std::strncmp("vt ", line.c_str(), 3) == 0) {
464  vector3 tex_coord;
465  std::sscanf(line.c_str() + 3, "%f%f", &tex_coord.x_, &tex_coord.y_);
466  tex_coords.push_back(tex_coord);
467  }
468 
469  // Normal
470  if (std::strncmp("vn ", line.c_str(), 3) == 0) {
471  vector3 normal;
472  std::sscanf(line.c_str() + 3, "%f%f%f", &normal.x_, &normal.y_, &normal.z_);
473  normals.push_back(normal);
474  }
475 
476  // Faces
477  if (std::strncmp("f ", line.c_str(), 2) == 0) {
478  uint32_t position_indices[3];
479  uint32_t tex_coord_indices[3];
480  uint32_t normal_indices[3];
481 
482  int32_t matches = std::sscanf(
483  line.c_str() + 2,
484  "%d/%d/%d %d/%d/%d %d/%d/%d\n",
485  &position_indices[0], &tex_coord_indices[0], &normal_indices[0],
486  &position_indices[1], &tex_coord_indices[1], &normal_indices[1],
487  &position_indices[2], &tex_coord_indices[2], &normal_indices[2]
488  );
489 
490  if (matches != 9) {
491  std::string msg = "Cannot load mesh [" + _file + "]. Make sure that the mesh is triangulated!";
492  log::error(msg);
493  return nullptr;
494  }
495 
496  // Add the Vertex.
497  for (uint32_t i = 0; i < 3; ++i) {
498  vertex vertex;
499 
500  vertex.position_ = positions[position_indices[i] - 1]; // -1 because .OBJ starts their index at 1.
501  vertex.normal_ = normals[normal_indices[i] - 1];
502  vertex.tex_coord_ = tex_coords[tex_coord_indices[i] - 1];
503 
504  vertices.push_back(vertex);
505  }
506  }
507  }
508 
509  // Index the vertices. (This defeats the point of indexing, since vertex data is still repeated, but creating a proper solution is too troublesome for now.)
510  for (uint32_t i = 0; i < vertices.size(); ++i) {
511  indices.push_back(i);
512  }
513 
514  // Set Tangents
515  for (uint32_t i = 0; i < indices.size(); i += 3) {
516  // This gives us 2 of the 3 edges.
517  matrix3x2 edges;
518  edges[0][0] = vertices[indices[i + 1]].position_.x_ - vertices[indices[i]].position_.x_;
519  edges[1][0] = vertices[indices[i + 1]].position_.y_ - vertices[indices[i]].position_.y_;
520  edges[2][0] = vertices[indices[i + 1]].position_.z_ - vertices[indices[i]].position_.z_;
521  edges[0][1] = vertices[indices[i + 2]].position_.x_ - vertices[indices[i]].position_.x_;
522  edges[1][1] = vertices[indices[i + 2]].position_.y_ - vertices[indices[i]].position_.y_;
523  edges[2][1] = vertices[indices[i + 2]].position_.z_ - vertices[indices[i]].position_.z_;
524 
525  // We have the length of 2 edges, and we know that each the edges can be represented by
526  // some length of the tangent + some length of the bitangent.
527  // One way we can find that length is by using the UV coordinates.
528  // So for example, we know that for one of the edges,
529  // the ratio of the edge along the tangent axis (also the same as the U axis for the texture coordinates) is U1 - U0 and.
530  // the ratio of the edge along the bitangent axis (also the same as the U axis for the texture coordinates) is V1 - V0.
531  matrix2x2 tangent_to_bitangent_ratio;
532  tangent_to_bitangent_ratio[0][0] = vertices[indices[i + 1]].tex_coord_.x_ - vertices[indices[i]].tex_coord_.x_;
533  tangent_to_bitangent_ratio[1][0] = vertices[indices[i + 1]].tex_coord_.y_ - vertices[indices[i]].tex_coord_.y_;
534  tangent_to_bitangent_ratio[0][1] = vertices[indices[i + 2]].tex_coord_.x_ - vertices[indices[i]].tex_coord_.x_;
535  tangent_to_bitangent_ratio[1][1] = vertices[indices[i + 2]].tex_coord_.y_ - vertices[indices[i]].tex_coord_.y_;
536 
537  matrix3x2 tangents = matrix_util::inverse_matrix(tangent_to_bitangent_ratio).value_or(matrix2x2::identity()) * edges;
538 
539  vertices[indices[i + 0]].tangent_ += vector3(tangents[0][0], tangents[1][0], tangents[2][0]);
540  vertices[indices[i + 1]].tangent_ += vector3(tangents[0][0], tangents[1][0], tangents[2][0]);
541  vertices[indices[i + 2]].tangent_ += vector3(tangents[0][0], tangents[1][0], tangents[2][0]);
542  }
543 
544  for (auto& vert: vertices) {
545  vert.tangent_.normalise();
546  }
547 
548  // Create the mesh.
549  meshes_[_name] = std::make_unique<mesh>(_name, vertices, indices);
550  return meshes_[_name].get();
551  }
