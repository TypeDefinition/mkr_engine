48  : name_{_name} {
49  mkr::log::info("creating shader program {}", _name.c_str());
50 
51  // Allocate uniform handle array.
52  uniform_handles_ = std::make_unique<GLint[]>(_num_uniforms);
53 
54  // Create the shader program.
55  program_handle_ = glCreateProgram();
56 
57  // Create the shaders.
58  std::vector<GLuint> vs_handles;
59  for (const auto& _vs_source: _vs_sources) {
60  // Create the fragment shader. OpenGL supports multiple vertex shaders per shader program.
61  GLuint handle = create_shader(GL_VERTEX_SHADER, _vs_source);
62  vs_handles.push_back(handle);
63  // Attach the shader to the program.
64  glAttachShader(program_handle_, handle);
65  }
66 
67  std::vector<GLuint> fs_handles;
68  for (const auto& _fs_source: _fs_sources) {
69  // Create the fragment shader. OpenGL supports multiple fragment shaders per shader program.
70  GLuint handle = create_shader(GL_FRAGMENT_SHADER, _fs_source);
71  fs_handles.push_back(handle);
72  // Attach the shader to the program.
73  glAttachShader(program_handle_, handle);
74  }
75 
76  // Link the shader program. Now that we have attached the shaders, this will use the attached shaders to create an executable that will run on the programmable vertex processor.
77  glLinkProgram(program_handle_);
78 
79  /**
80  * Now that we are done creating the shader program, we no longer need the shaders, and they can be deleted.
81  * It is also possible to store the shaders to create other shader programs,
82  * but there isn't a compelling reason to do so since we can just re-create them again if necessary. */
83  for (GLuint vs_handle: vs_handles) {
84  glDetachShader(program_handle_, vs_handle);
85  glDeleteShader(vs_handle);
86  }
87  for (GLuint fs_handle: fs_handles) {
88  glDetachShader(program_handle_, fs_handle);
89  glDeleteShader(fs_handle);
90  }
91 
92  // Verify link status.
93  GLint status = 0;
94  glGetProgramiv(program_handle_, GL_LINK_STATUS, &status);
95  if (status == GL_FALSE) {
96  GLint log_length = 0;
97  glGetProgramiv(program_handle_, GL_INFO_LOG_LENGTH, &log_length);
98  std::unique_ptr<GLchar> info_log{new GLchar[log_length]};
99  glGetProgramInfoLog(program_handle_, log_length, &log_length, info_log.get());
100  mkr::log::error(info_log.get());
101  throw std::runtime_error(info_log.get());
102  }
103 
104  mkr::log::info("shader program {} created", _name.c_str());
105  }
