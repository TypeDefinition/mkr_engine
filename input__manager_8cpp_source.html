23  button_handler_.on_key_up(mask);
24  axis_handler_.on_key_up(mask);
25  }
26  break;
27

29  case SDL_MOUSEBUTTONDOWN: {
30  // Mouse defaults to controller_index_0.
31  const input_mask_t mask = input_helper::get_input_mask(input_context_, controller_index_0, sdl_to_keycode::from_mouse_button(e.button.button));
32  button_handler_.on_key_down(mask);
33  click_handler_.on_key_down(mask);
34  }
35  break;
36  case SDL_MOUSEBUTTONUP: {
37  // Mouse defaults to controller_index_0.
38  const input_mask_t mask = input_helper::get_input_mask(input_context_, controller_index_0, sdl_to_keycode::from_mouse_button(e.button.button));
39  button_handler_.on_key_up(mask);
40  click_handler_.on_key_up(mask);
41  }
42  break;
43  case SDL_MOUSEMOTION: {
44  // Mouse defaults to controller_index_0.
45  const vector2 pos{(float) e.motion.x, (float) e.motion.y};
46  const vector2 delta{(float) e.motion.xrel, (float) e.motion.yrel};
47 
48  // Clicks
49  if (e.motion.state & SDL_BUTTON_LMASK) {
50  click_handler_.on_motion(input_helper::get_input_mask(input_context_, controller_index_0, kc_mouse_left), pos);
51  }
52  if (e.motion.state & SDL_BUTTON_RMASK) {
53  click_handler_.on_motion(input_helper::get_input_mask(input_context_, controller_index_0, kc_mouse_right), pos);
54  }
55  if (e.motion.state & SDL_BUTTON_MMASK) {
56  click_handler_.on_motion(input_helper::get_input_mask(input_context_, controller_index_0, kc_mouse_middle), pos);
57  }
58  if (e.motion.state & SDL_BUTTON_X1MASK) {
59  click_handler_.on_motion(input_helper::get_input_mask(input_context_, controller_index_0, kc_mouse_x1), pos);
60  }
61  if (e.motion.state & SDL_BUTTON_X2MASK) {
62  click_handler_.on_motion(input_helper::get_input_mask(input_context_, controller_index_0, kc_mouse_x2), pos);
63  }
64 
65  // Axis
66  axis_handler_.on_axis(input_helper::get_input_mask(input_context_, controller_index_0, kc_mouse_axis_x), delta.x_);
67  axis_handler_.on_axis(input_helper::get_input_mask(input_context_, controller_index_0, kc_mouse_axis_y), delta.y_);
68 
69  // Motion
70  motion_handler_.on_motion(input_helper::get_input_mask(input_context_, controller_index_0, kc_mouse_motion), pos, delta);
71  }
72  break;
73 
74  // Gamepad
75  case SDL_CONTROLLERAXISMOTION: {
76  // Normalise the axis value to between [-1, 1]. SDL gamepad axis values range is [-32768, 32767].
77  const auto value = e.caxis.value < 0 ? (float) e.caxis.value / -32768.0f : (float) e.caxis.value / 32767.0f;
78  const input_mask_t mask = input_helper::get_input_mask(input_context_,
79  sdl_to_controller_index::from_joystick_id(e.caxis.which),

81  axis_handler_.on_axis(mask, value);
82  }
83  break;
84  case SDL_CONTROLLERBUTTONDOWN: {
85  const input_mask_t mask = input_helper::get_input_mask(input_context_,
86  sdl_to_controller_index::from_joystick_id(e.cbutton.which),
87  sdl_to_keycode::from_gamepad_button((SDL_GameControllerButton) e.cbutton.button));
88  button_handler_.on_key_down(mask);
89  }
90  break;
91  case SDL_CONTROLLERBUTTONUP: {
92  const input_mask_t mask = input_helper::get_input_mask(input_context_,
93  sdl_to_controller_index::from_joystick_id(e.cbutton.which),
94  sdl_to_keycode::from_gamepad_button((SDL_GameControllerButton) e.cbutton.button));
95  button_handler_.on_key_up(mask);
96  }
97  break;
98  default:
99  break;
