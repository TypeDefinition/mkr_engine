108  /**
109  * IMPORTANT: Unlike glTexImage2D, we have to explicitly state the number of mipmaps to generate.
110  * We can no longer leave it at 0 and expect OpenGL to figure out how many mipmaps to generate.
111  * This is because glTextureStorage2D creates a fixed storage space for our texture, not like
112  * glTexImage2D where it can resize itself to generate mipmaps. The glTextureStorage2D documentation
113  * states "GL_INVALID_VALUE is generated if width, height or levels are less than 1." */
114  // Direct-State-Access Method (OpenGL 4.5) Create Texture
115  glCreateTextures(GL_TEXTURE_2D, 1, &handle_);
116 
117  glTextureStorage2D(handle_, (GLsizei)mip_map_level(width_, height_),
118  (GLenum)_format, ///< Format to store the texture data in OpenGL.
119  (GLsizei)width_, (GLsizei)height_);
120  glTextureSubImage2D(handle_, 0, 0, 0, (GLsizei)width_, (GLsizei)height_,
121  (GLenum)pixel_format::sized_to_base(_format), ///< Format of the image data.
122  GL_UNSIGNED_BYTE, _data);
