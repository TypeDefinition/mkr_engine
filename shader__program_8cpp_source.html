218  shader_program::shader_program(const std::string& _name, render_pass _render_pass, const std::vector<std::string>& _vs_sources, const std::vector<std::string>& _fs_sources)
219  : name_{_name}, render_pass_{_render_pass} {
220  // Create the shader program.
221  program_handle_ = glCreateProgram();
222 
223  // Create the shaders.
224  std::vector<GLuint> vs_handles;
225  for (const auto& _vs_source: _vs_sources) {
226  // Create the fragment shader. OpenGL supports multiple vertex shaders per shader program.
227  GLuint handle = create_shader(GL_VERTEX_SHADER, _vs_source);
228  vs_handles.push_back(handle);
229  // Attach the shader to the program.
230  glAttachShader(program_handle_, handle);
231  }
232 
233  std::vector<GLuint> fs_handles;
234  for (const auto& _fs_source: _fs_sources) {
235  // Create the fragment shader. OpenGL supports multiple fragment shaders per shader program.
236  GLuint handle = create_shader(GL_FRAGMENT_SHADER, _fs_source);
237  fs_handles.push_back(handle);
238  // Attach the shader to the program.
239  glAttachShader(program_handle_, handle);
240  }
241 
242  // Link the shader program. Now that we have attached the shaders, this will use the attached shaders to create an executable that will run on the programmable vertex processor.
243  glLinkProgram(program_handle_);
244 
245  /**
246  * Now that we are done creating the shader program, we no longer need the shaders, and they can be deleted.
247  * It is also possible to store the shaders to create other shader programs,
248  * but there isn't a compelling reason to do so since we can just re-create them again if necessary. */
249  for (GLuint vs_handle: vs_handles) {
250  glDetachShader(program_handle_, vs_handle);
251  glDeleteShader(vs_handle);
252  }
253  for (GLuint fs_handle: fs_handles) {
254  glDetachShader(program_handle_, fs_handle);
255  glDeleteShader(fs_handle);
256  }
257 
258  // Verify link status.
259  GLint status = 0;
260  glGetProgramiv(program_handle_, GL_LINK_STATUS, &status);
261  if (status == GL_FALSE) {
262  GLint log_length = 0;
263  glGetProgramiv(program_handle_, GL_INFO_LOG_LENGTH, &log_length);
264  std::shared_ptr<GLchar> info_log{new GLchar[log_length]};
265  glGetProgramInfoLog(program_handle_, log_length, &log_length, info_log.get());
266  mkr::log::error(info_log.get());
267  throw std::runtime_error(info_log.get());
268  }
269 
270  mkr::log::info("shader program {} created", _name.c_str());
271 
272  assign_uniforms();
273  assign_textures();
274  }
