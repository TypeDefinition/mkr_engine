229  static std::unique_ptr<mesh> load_obj(const std::string &_name, const std::string &_file) {
230  // Vertex Data and Index Data
231  std::vector<vertex> vertices;
232  std::vector<uint32_t> indices;
233 
234  // Vertex Attribute(s)
235  std::vector<vector3> positions;
236  std::vector<vector3> normals;
237  std::vector<vector2> tex_coords;
238 
239  // Read the file.
240  std::string file_str = file_util::file_to_str(_file);
241  std::stringstream ss(file_str.c_str());
242  while (ss.good()) {
243  std::string line;
244  std::getline(ss, line, '\n');
245 
246  // Position
247  if (std::strncmp("v ", line.c_str(), 2) == 0) {
248  vector3 position;
249  std::sscanf(line.c_str() + 2, "%f%f%f", &position.x_, &position.y_, &position.z_);
250  positions.push_back(position);
251  }
252 
253  // TexCoord
254  if (std::strncmp("vt ", line.c_str(), 3) == 0) {
255  vector2 tex_coord;
256  std::sscanf(line.c_str() + 3, "%f%f", &tex_coord.x_, &tex_coord.y_);
257  tex_coords.push_back(tex_coord);
258  }
259 
260  // Normal
261  if (std::strncmp("vn ", line.c_str(), 3) == 0) {
262  vector3 normal;
263  std::sscanf(line.c_str() + 3, "%f%f%f", &normal.x_, &normal.y_, &normal.z_);
264  normals.push_back(normal);
265  }
266 
267  // Faces
268  if (std::strncmp("f ", line.c_str(), 2) == 0) {
269  uint32_t position_indices[3];
270  uint32_t tex_coord_indices[3];
271  uint32_t normal_indices[3];
272 
273  int32_t matches = std::sscanf(
274  line.c_str() + 2,
275  "%d/%d/%d %d/%d/%d %d/%d/%d\n",
276  &position_indices[0], &tex_coord_indices[0], &normal_indices[0],
277  &position_indices[1], &tex_coord_indices[1], &normal_indices[1],
278  &position_indices[2], &tex_coord_indices[2], &normal_indices[2]
279  );
280 
281  if (matches != 9) {
282  std::string msg = "Cannot load mesh [" + _file + "]. Make sure that the mesh is triangulated!";
283  MKR_CORE_ERROR(msg);
284  return nullptr;
285  }
286 
287  // Add the Vertex.
288  for (uint32_t i = 0; i < 3; ++i) {
289  vertex vertex;
290 
291  vertex.position_ = positions[position_indices[i] - 1]; // -1 because .OBJ starts their index at 1.
292  vertex.normal_ = normals[normal_indices[i] - 1];
293  vertex.tex_coord_ = tex_coords[tex_coord_indices[i] - 1];
294 
295  vertices.push_back(vertex);
296  }
297  }
298  }
299 
300  // Index the vertices. (This defeats the point of indexing, since vertex data is still repeated, but creating a proper solution is too troublesome for now.)
301  for (uint32_t i = 0; i < vertices.size(); ++i) {
302  indices.push_back(i);
303  }
304 
305  // Set Tangents
306  for (uint32_t i = 0; i < indices.size(); i += 3) {
307  // This gives us 2 of the 3 edges.
308  matrix3x2 edges;
309  edges[0][0] = vertices[indices[i + 1]].position_.x_ - vertices[indices[i]].position_.x_;
310  edges[1][0] = vertices[indices[i + 1]].position_.y_ - vertices[indices[i]].position_.y_;
311  edges[2][0] = vertices[indices[i + 1]].position_.z_ - vertices[indices[i]].position_.z_;
312  edges[0][1] = vertices[indices[i + 2]].position_.x_ - vertices[indices[i]].position_.x_;
313  edges[1][1] = vertices[indices[i + 2]].position_.y_ - vertices[indices[i]].position_.y_;
314  edges[2][1] = vertices[indices[i + 2]].position_.z_ - vertices[indices[i]].position_.z_;
315 
316  /* We have the length of 2 edges, and we know that each the edges can be represented by some length of the tangent + some length of the bitangent.
317  One way we can find that length is by using the UV coordinates.
318  So for example, we know that for one of the edges,
319  the ratio of the edge along the tangent axis (also the same as the U axis for the texture coordinates) is U1 - U0 and.
320  the ratio of the edge along the bitangent axis (also the same as the V axis for the texture coordinates) is V1 - V0. */
321  matrix2x2 uv_delta;
322  uv_delta[0][0] = vertices[indices[i + 1]].tex_coord_.x_ - vertices[indices[i]].tex_coord_.x_;
323  uv_delta[1][0] = vertices[indices[i + 1]].tex_coord_.y_ - vertices[indices[i]].tex_coord_.y_;
324  uv_delta[0][1] = vertices[indices[i + 2]].tex_coord_.x_ - vertices[indices[i]].tex_coord_.x_;
325  uv_delta[1][1] = vertices[indices[i + 2]].tex_coord_.y_ - vertices[indices[i]].tex_coord_.y_;
326 
327  matrix3x2 tangents = matrix_util::inverse_matrix(uv_delta).value_or(matrix2x2::identity()) * edges;
328 
329  vertices[indices[i + 0]].tangent_ += vector3(tangents[0][0], tangents[1][0], tangents[2][0]);
330  vertices[indices[i + 1]].tangent_ += vector3(tangents[0][0], tangents[1][0], tangents[2][0]);
331  vertices[indices[i + 2]].tangent_ += vector3(tangents[0][0], tangents[1][0], tangents[2][0]);
332  }
333 
334  for (auto& vert: vertices) {
335  vert.tangent_.normalise();
336  }
337 
338  // Create the mesh.
339  return std::make_unique<mesh>(_name, vertices, indices);
340  }
