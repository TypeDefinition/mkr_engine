90  void graphics_renderer::render() {
91  // Shadow maps for spot and point lights can be shared between cameras.
92  const auto num_lights = maths_util::min<int32_t>(lights_.size(), lighting::max_lights);
93  for (auto i = 0; i < num_lights; ++i) {
94  const auto& light = lights_[i].light_;
95  const auto& trans = lights_[i].transform_;
96  if (light.get_mode() == light_mode::point) {
97  light_view_projection_matrix_[i] = point_shadow(scube_buff_[i].get(), trans, light);
98  }
99  else if (light.get_mode() == light_mode::spot) {
100  light_view_projection_matrix_[i] = spot_shadow(s2d_buff_[i].get(), trans, light);
101  }
102  }
103 
104  // Render the scene once for every camera.
105  while (!cameras_.empty()) {
106  const auto& cam = cameras_.top().camera_;
107  const auto& trans = cameras_.top().transform_;
108 
109  // Shadow maps for directional lights need to be recalculated once for each camera.
110  for (auto i = 0; i < num_lights; ++i) {
111  const auto& light = lights_[i].light_;
112  if (light.get_mode() == light_mode::directional) {
113  light_view_projection_matrix_[i] = directional_shadow(s2d_buff_[i].get(), lights_[i].transform_, trans, cam);
114  }
115  }
116 
117  // In OpenGL convention, the camera looks down the -z axis.
118  const auto& view_dir_x = -trans.left_;
119  const auto& view_dir_y = trans.up_;
120  const auto& view_dir_z = -trans.forward_;
121 
122  // View Matrix
123  const auto view_matrix = matrix_util::view_matrix(trans.position_, trans.forward_, trans.up_);
124  // Projection Matrix
125  const auto projection_matrix = (cam.mode_ == projection_mode::perspective)
126  ? matrix_util::perspective_matrix(cam.aspect_ratio_, cam.fov_, cam.near_plane_, cam.far_plane_)
127  : matrix_util::orthographic_matrix(cam.aspect_ratio_, cam.ortho_size_, cam.near_plane_, cam.far_plane_);
128  const auto inv_view_matrix = matrix_util::inverse_matrix(view_matrix).value_or(matrix4x4 ::identity());
129 
130  // Render forward pass.
131  forward_pass(view_matrix, projection_matrix, inv_view_matrix, view_dir_z, view_dir_y, view_dir_x);
132 
133  // Pop camera off the priority queue.
134  cameras_.pop();
135  }
136 
137  // Clear objects for next frame.
138  lights_.clear();
139  deferred_meshes_.clear();
140  forward_opaque_meshes_.clear();
141  forward_transparent_meshes_.clear();
